Data Wrangling

Task

Extract data of the overseas residents’ visits to Japan from Japan Tourism Marketing Co.’s website. Format the data as tidy data (see Wickham 2014), and make sure it is suitable for time series analyses. Export the data as a csv file.

Prerequisites

# Setup virtual environment and install packages
# Run in terminal
# python3 -m venv env
# source env/bin/activate
# pip install pandas openpyxl

# Record requirements
# pip3 freeze > forecasting/jp_tourism/requirements.txt
# Load required packages
import pandas as pd
from urllib.request import urlretrieve

Preparing variables

# Prepare the date column for monthly data (1996 - 2023)
start_date = pd.to_datetime("1996-01")
end_date = pd.to_datetime("2023-12")
date_range = pd.date_range(start_date, end_date, freq = "MS").date
period = len(date_range)
date = pd.Series(date_range)
# Prepare empty DataFrame with Date, Region, Purpose, Visits column
column_types = {"Date": "object", "Region": "string", "Purpose": "string", "Visits": "int"}
jp_visits  = pd.DataFrame(columns = "Date Region Purpose Visits".split()).astype(column_types)
# Prepare the regions column and location dictionary
locations = {}
regions = ["Asia", "Europe", "North America", "South America", "Africa", "Oceania", "Stateless"]
for region in regions:
  locations[region] = pd.Series([region for x in range(period)], dtype = "string")

Data Extraction

# Download the dataset
# Retrieved on June 26, 2024
urlretrieve("https://www.tourism.jp/wp/wp-content/uploads/2024/06/JTM_inbound_20240529eng.xlsx", "JTM_inbound_20240529eng.xlsx")

For each region, I will extract the data from the Excel file and structure it as defined in the empty DataFrame created earlier. I chose to merge the visit purpose “Others” and “Short Excursion” as the “Short Excursion” classification has not been used in the later dates.

Asia & Europe

for location in "Asia Europe".split():
  df = pd.read_excel("JTM_inbound_20240529eng.xlsx", 
  sheet_name = location,
  header = [3, 4],
  nrows = period
  )
  df = df.fillna(value = 0)
  for type in "Tourist Business Others".split():
    purpose = pd.Series([type for x in range(period)], dtype = "string")
    if type != "Others":
      visits = df.loc[:, (f'{location} Total', type)].astype('int')
    else:
      visits = df.loc[:, (f'{location} Total', 'Others')] + df.loc[:, (f'{location} Total', 'Short Excursion')].astype('int')
    data_dict = {"Date": date, "Region": locations[location], "Purpose": purpose, "Visits": visits}
    data = pd.concat(data_dict, axis = 1)
    jp_visits = pd.concat([jp_visits, data], axis = 0)

Others & Grand Total

df = pd.read_excel("JTM_inbound_20240529eng.xlsx", 
  sheet_name = "Others & Grand Total",
  header = [3, 4],
  nrows = period
  )
  df = df.fillna(value = 0)
for location in ["North America", "South America", "Africa", "Oceania", "Stateless"]:
  for type in "Tourist Business Others".split():
    purpose = pd.Series([type for x in range(period)], dtype = "string")
    if location != "Stateless":
      if type != "Others":
        visits = df.loc[:, (f'{location} Total', type)].astype('int')
      else:
        visits = df.loc[:, (f'{location} Total', 'Others')] + df.loc[:, (f'{location} Total', 'Short Excursion')].astype('int')
    else:
      if type != "Others":
        visits = df.loc[:, (f'{location}', type)].astype('int')
      else:
        visits = df.loc[:, (f'{location}', 'Others')] + df.loc[:, (f'{location}', 'Short Excursion')].astype('int')
    data_dict = {"Date": date, "Region": locations[location], "Purpose": purpose, "Visits": visits}
    data = pd.concat(data_dict, axis = 1)
    jp_visits = pd.concat([jp_visits, data], axis = 0)

Export data

jp_visits.to_csv("jp_visits.csv", index = False)
